```python
"""
AMAZON PRODUCT PROFITABILITY ANALYSIS - PYTHON ETL
Business Goal: Clean 1,465 product records, calculate true profitability
Tools in Service of Business: Python helps us understand product economics
"""

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# ============================================
# 1. LOAD DATA - 1,465 products
# ============================================
print("🚀 Loading Amazon product data...")
df = pd.read_csv('amazon_products.csv')
print(f"📊 Dataset Shape: {df.shape}")  # (1,465, 15)
print(f"📊 Columns: {df.columns.tolist()}")

# ============================================
# 2. DATA CLEANING - Because dirty data = wrong decisions
# ============================================
print("\n🧹 Cleaning data for accurate business insights...")

# Check for missing values
print(f"❓ Missing values before cleaning:")
print(df.isnull().sum())

# Handle missing values
# For numeric columns: fill with median
numeric_cols = ['price', 'cost', 'discount_percentage', 'shipping_cost', 'returns']
for col in numeric_cols:
    df[col].fillna(df[col].median(), inplace=True)

# For categorical: fill with mode
categorical_cols = ['category', 'subcategory', 'seller', 'fulfillment_type']
for col in categorical_cols:
    df[col].fillna(df[col].mode()[0], inplace=True)

# Remove duplicates
initial_count = len(df)
df.drop_duplicates(subset=['product_id', 'asin'], inplace=True)
dupes_removed = initial_count - len(df)
print(f"✅ Removed {dupes_removed} duplicate products")

# Remove outliers (extreme values that don't make business sense)
df = df[df['price'] > 0]  # Price must be positive
df = df[df['cost'] > 0]    # Cost must be positive
df = df[df['discount_percentage'].between(0, 100)]  # Discount between 0-100%

# ============================================
# 3. FEATURE ENGINEERING - Creating profitability metrics
# ============================================
print("\n⚙️ Creating profitability metrics...")

# Core financial calculations
df['revenue'] = df['price'] * df['units_sold']
df['total_cost'] = (df['cost'] * df['units_sold']) + df['shipping_cost'] + (df['returns'] * df['cost'])
df['profit'] = df['revenue'] - df['total_cost']
df['margin_percentage'] = (df['profit'] / df['revenue'] * 100).round(2)

# Create profit/loss flag
df['profit_loss_status'] = np.where(df['profit'] > 0, 'Profitable', 'Loss')
df['loss_amount'] = np.where(df['profit'] < 0, abs(df['profit']), 0)

# Create price segments
df['price_segment'] = pd.cut(
    df['price'],
    bins=[0, 500, 2000, 5000, 100000],
    labels=['Budget (<₹500)', 'Mid-Range (₹500-2000)', 
            'Premium (₹2000-5000)', 'Luxury (>₹5000)']
)

# Discount impact analysis
df['discount_impact'] = pd.cut(
    df['discount_percentage'],
    bins=[0, 10, 20, 30, 50, 100],
    labels=['0-10%', '10-20%', '20-30%', '30-50%', '>50%']
)

# ROI calculation (Return on Investment)
df['roi'] = (df['profit'] / df['total_cost'] * 100).round(2)

# ============================================
# 4. BUSINESS INSIGHTS - Quick Python analysis
# ============================================
print("\n💡 Quick Business Insights from Python:")

# Overall profitability
total_profit = df['profit'].sum()
total_revenue = df['revenue'].sum()
total_loss = df['loss_amount'].sum()
profitable_count = len(df[df['profit'] > 0])
loss_count = len(df[df['profit'] < 0])

print(f"💰 Total Profit: ₹{total_profit:,.2f}")
print(f"📉 Total Loss: ₹{total_loss:,.2f}")
print(f"✅ Profitable Products: {profitable_count} ({profitable_count/len(df)*100:.1f}%)")
print(f"❌ Loss-Making Products: {loss_count} ({loss_count/len(df)*100:.1f}%)")

# Profitability by price segment
segment_profit = df.groupby('price_segment').agg({
    'profit': 'sum',
    'product_id': 'count',
    'margin_percentage': 'mean'
}).round(2)
segment_profit.columns = ['Total Profit', 'Product Count', 'Avg Margin %']
print("\n📊 Profitability by Price Segment:")
print(segment_profit.sort_values('Total Profit', ascending=False))

# Discount analysis
discount_analysis = df.groupby('discount_impact').agg({
    'profit': 'mean',
    'margin_percentage': 'mean',
    'product_id': 'count'
}).round(2)
print("\n🏷️ Discount Impact on Profitability:")
print(discount_analysis.sort_index())

# Category performance
category_performance = df.groupby('category').agg({
    'profit': 'sum',
    'margin_percentage': 'mean',
    'units_sold': 'sum'
}).round(2)
print("\n📦 Category Performance:")
print(category_performance.sort_values('profit', ascending=False))

# Loss-making products deep dive
loss_products = df[df['profit'] < 0].copy()
print("\n🔍 Loss-Making Products Summary:")
print(f"Count: {len(loss_products)}")
print(f"Avg Price: ₹{loss_products['price'].mean():,.2f}")
print(f"Avg Cost: ₹{loss_products['cost'].mean():,.2f}")
print(f"Avg Discount: {loss_products['discount_percentage'].mean():.1f}%")
print(f"Total Loss: ₹{loss_products['loss_amount'].sum():,.2f}")

# ============================================
# 5. VISUALIZE - Quick insights before Power BI
# ============================================
fig, axes = plt.subplots(2, 2, figsize=(15, 10))

# Profit by price segment
segment_data = segment_profit['Total Profit'].sort_values()
axes[0,0].barh(segment_data.index, segment_data.values, color='#FF9900')
axes[0,0].set_title('Total Profit by Price Segment', fontsize=14)
axes[0,0].set_xlabel('Profit (₹)')
axes[0,0].axvline(x=0, color='black', linestyle='-', alpha=0.3)

# Discount vs Margin
axes[0,1].scatter(df['discount_percentage'], df['margin_percentage'], 
                  alpha=0.5, color='#FF9900', s=50)
axes[0,1].set_title('Discount % vs Margin %', fontsize=14)
axes[0,1].set_xlabel('Discount Percentage')
axes[0,1].set_ylabel('Margin %')
axes[0,1].axhline(y=0, color='red', linestyle='--', alpha=0.5)

# Loss products distribution
loss_by_category = loss_products['category'].value_counts()
axes[1,0].bar(loss_by_category.index[:5], loss_by_category.values[:5], color='#FF9900')
axes[1,0].set_title('Top 5 Categories with Loss Products', fontsize=14)
axes[1,0].set_xlabel('Category')
axes[1,0].set_ylabel('Number of Loss Products')
axes[1,0].tick_params(axis='x', rotation=45)

# Margin distribution
axes[1,1].hist(df['margin_percentage'], bins=30, color='#FF9900', edgecolor='black')
axes[1,1].set_title('Margin Distribution', fontsize=14)
axes[1,1].set_xlabel('Margin %')
axes[1,1].set_ylabel('Frequency')
axes[1,1].axvline(x=0, color='red', linestyle='--', linewidth=2)

plt.tight_layout()
plt.savefig('python_profitability_insights.png', dpi=100)
print("\n✅ Quick insights visualization saved!")

# ============================================
# 6. EXPORT CLEAN DATA - Ready for SQL, AWS, Power BI
# ============================================
df.to_csv('amazon_products_clean.csv', index=False)
print("\n✅ Clean data exported for SQL/AWS/Power BI")

# Business Summary
print("\n" + "="*60)
print("🎯 BUSINESS SUMMARY (from Python)")
print("="*60)
print(f"📦 Total Products Analyzed: {len(df):,}")
print(f"💰 Total Revenue: ₹{df['revenue'].sum():,.2f}")
print(f"💰 Total Cost: ₹{df['total_cost'].sum():,.2f}")
print(f"📈 Total Profit: ₹{total_profit:,.2f}")
print(f"📊 Avg Margin: {df['margin_percentage'].mean():.2f}%")
print(f"❌ Loss-Making Products: {loss_count} (₹{total_loss:,.2f} loss)")
print(f"🎯 Best Performing Segment: {segment_profit['Total Profit'].idxmax()}")
print(f"⚠️ Critical Issue: Loss products have {loss_products['discount_percentage'].mean():.1f}% avg discount")
print("="*60)
```

```sql
/*
AMAZON PRODUCT PROFITABILITY ANALYSIS - SQL SERVER QUERIES
Business Goal: Answer specific business questions about product profitability
Focus: Product segments, loss analysis, discount optimization
*/

-- ============================================
-- 1. DATABASE SETUP - Product Data Warehouse
-- ============================================
CREATE DATABASE AmazonProfitability;
GO

USE AmazonProfitability;
GO

-- Main products table
CREATE TABLE Products (
    ProductID INT PRIMARY KEY,
    ASIN VARCHAR(20),
    ProductName NVARCHAR(200),
    Category VARCHAR(50),
    Subcategory VARCHAR(50),
    Price DECIMAL(10,2),
    Cost DECIMAL(10,2),
    DiscountPercentage DECIMAL(5,2),
    UnitsSold INT,
    Returns INT,
    ShippingCost DECIMAL(10,2),
    Revenue DECIMAL(12,2),
    TotalCost DECIMAL(12,2),
    Profit DECIMAL(12,2),
    MarginPercentage AS (Profit / NULLIF(Revenue, 0) * 100),
    PriceSegment VARCHAR(20),
    ProfitLossStatus VARCHAR(10),
    CreatedDate DATETIME DEFAULT GETDATE()
);

-- Load data (from Python cleaned file)
BULK INSERT Products
FROM 'C:\data\amazon_products_clean.csv'
WITH (
    FIRSTROW = 2,
    FIELDTERMINATOR = ',',
    ROWTERMINATOR = '\n',
    TABLOCK
);

-- ============================================
-- 2. BUSINESS QUESTION: Overall profitability snapshot
-- ============================================
SELECT 
    COUNT(*) AS TotalProducts,
    SUM(CASE WHEN Profit > 0 THEN 1 ELSE 0 END) AS ProfitableProducts,
    SUM(CASE WHEN Profit < 0 THEN 1 ELSE 0 END) AS LossProducts,
    CAST(100.0 * SUM(CASE WHEN Profit > 0 THEN 1 ELSE 0 END) / COUNT(*) AS DECIMAL(5,1)) AS ProfitablePercent,
    SUM(Revenue) AS TotalRevenue,
    SUM(TotalCost) AS TotalCost,
    SUM(Profit) AS TotalProfit,
    AVG(MarginPercentage) AS AvgMargin
FROM Products;

-- ============================================
-- 3. BUSINESS QUESTION: Which price segments perform best?
-- ============================================
SELECT 
    PriceSegment,
    COUNT(*) AS ProductCount,
    SUM(Revenue) AS TotalRevenue,
    SUM(Profit) AS TotalProfit,
    AVG(MarginPercentage) AS AvgMargin,
    SUM(CASE WHEN Profit < 0 THEN 1 ELSE 0 END) AS LossCount,
    CAST(100.0 * SUM(CASE WHEN Profit < 0 THEN 1 ELSE 0 END) / COUNT(*) AS DECIMAL(5,1)) AS LossRate,
    -- Performance rating
    CASE 
        WHEN AVG(MarginPercentage) > 10 THEN '⭐ STAR'
        WHEN AVG(MarginPercentage) BETWEEN 5 AND 10 THEN '✅ GOOD'
        WHEN AVG(MarginPercentage) BETWEEN 0 AND 5 THEN '🟡 AVERAGE'
        ELSE '🔴 LOSS'
    END AS PerformanceRating
FROM Products
GROUP BY PriceSegment
ORDER BY TotalProfit DESC;

-- ============================================
-- 4. BUSINESS QUESTION: Discount impact on profitability
-- ============================================
WITH DiscountBands AS (
    SELECT 
        CASE 
            WHEN DiscountPercentage < 10 THEN '1. 0-10%'
            WHEN DiscountPercentage BETWEEN 10 AND 20 THEN '2. 10-20%'
            WHEN DiscountPercentage BETWEEN 20 AND 30 THEN '3. 20-30%'
            WHEN DiscountPercentage BETWEEN 30 AND 50 THEN '4. 30-50%'
            ELSE '5. >50%'
        END AS DiscountBand,
        Profit,
        MarginPercentage,
        Revenue,
        Price
    FROM Products
)
SELECT 
    DiscountBand,
    COUNT(*) AS ProductCount,
    AVG(Profit) AS AvgProfit,
    AVG(MarginPercentage) AS AvgMargin,
    SUM(CASE WHEN Profit < 0 THEN 1 ELSE 0 END) AS LossCount,
    CAST(100.0 * SUM(CASE WHEN Profit < 0 THEN 1 ELSE 0 END) / COUNT(*) AS DECIMAL(5,1)) AS LossRate,
    -- Optimal discount recommendation
    CASE 
        WHEN AVG(MarginPercentage) > 10 THEN 'Optimal'
        WHEN AVG(MarginPercentage) > 5 THEN 'Acceptable'
        WHEN AVG(MarginPercentage) > 0 THEN 'Reduce Discount'
        ELSE 'Stop Discounting'
    END AS Recommendation
FROM DiscountBands
GROUP BY DiscountBand
ORDER BY DiscountBand;

-- ============================================
-- 5. BUSINESS QUESTION: Loss-making products deep dive
-- ============================================
WITH LossProducts AS (
    SELECT 
        ProductID,
        ProductName,
        Category,
        Price,
        Cost,
        DiscountPercentage,
        UnitsSold,
        Profit,
        MarginPercentage,
        -- Compare to category averages
        AVG(Price) OVER (PARTITION BY Category) AS AvgCategoryPrice,
        AVG(DiscountPercentage) OVER (PARTITION BY Category) AS AvgCategoryDiscount
    FROM Products
    WHERE Profit < 0
)
SELECT 
    ProductName,
    Category,
    Price,
    Cost,
    DiscountPercentage,
    UnitsSold,
    Profit,
    -- Why is this product losing money?
    CASE 
        WHEN Price < AvgCategoryPrice * 0.7 THEN 'Priced too low'
        WHEN DiscountPercentage > AvgCategoryDiscount * 1.5 THEN 'Over-discounted'
        WHEN Cost > Price * 0.8 THEN 'Cost too high'
        ELSE 'Multiple factors'
    END AS LossReason,
    -- Recommended action
    CASE 
        WHEN Price < AvgCategoryPrice * 0.7 THEN 'Increase price by 20%'
        WHEN DiscountPercentage > AvgCategoryDiscount * 1.5 THEN 'Reduce discount by half'
        WHEN Cost > Price * 0.8 THEN 'Negotiate with supplier'
        ELSE 'Comprehensive review needed'
    END AS RecommendedAction
FROM LossProducts
ORDER BY Profit;

-- ============================================
-- 6. BUSINESS QUESTION: Category performance ranking
-- ============================================
SELECT 
    Category,
    COUNT(*) AS ProductCount,
    SUM(Revenue) AS TotalRevenue,
    SUM(Profit) AS TotalProfit,
    AVG(MarginPercentage) AS AvgMargin,
    SUM(CASE WHEN Profit < 0 THEN 1 ELSE 0 END) AS LossCount,
    -- Rank categories by profitability
    RANK() OVER (ORDER BY SUM(Profit) DESC) AS ProfitRank,
    RANK() OVER (ORDER BY AVG(MarginPercentage) DESC) AS MarginRank
FROM Products
GROUP BY Category
ORDER BY TotalProfit DESC;

-- ============================================
-- 7. STORED PROCEDURE: Loss Product Recovery Plan
-- ============================================
CREATE PROCEDURE sp_GetLossProductRecovery
    @MinLossAmount DECIMAL(10,2) = 0
AS
BEGIN
    SELECT 
        ProductID,
        ProductName,
        Category,
        Price,
        Cost,
        DiscountPercentage,
        Profit AS CurrentLoss,
        -- Recovery strategies
        Price * 1.2 AS RecommendedPrice,  -- 20% price increase
        DiscountPercentage * 0.5 AS RecommendedDiscount,  -- 50% discount reduction
        (Price * 1.2 * UnitsSold) - (Cost * UnitsSold + ShippingCost) AS ProjectedProfit,
        -- ROI calculation
        CASE 
            WHEN Price * 0.2 * UnitsSold > 5000 THEN 'HIGH PRIORITY'
            WHEN Price * 0.2 * UnitsSold > 1000 THEN 'MEDIUM PRIORITY'
            ELSE 'LOW PRIORITY'
        END AS Priority
    FROM Products
    WHERE Profit < 0
        AND ABS(Profit) >= @MinLossAmount
    ORDER BY Profit;
END;
GO

-- Execute: EXEC sp_GetLossProductRecovery 1000;

-- ============================================
-- 8. BUSINESS VIEW - For Power BI Dashboard
-- ============================================
CREATE VIEW vw_ProfitabilityDashboard AS
SELECT 
    ProductID,
    ProductName,
    Category,
    Subcategory,
    Price,
    Cost,
    DiscountPercentage,
    UnitsSold,
    Revenue,
    TotalCost,
    Profit,
    MarginPercentage,
    PriceSegment,
    ProfitLossStatus,
    -- Profit categories
    CASE 
        WHEN MarginPercentage > 15 THEN 'High Profit'
        WHEN MarginPercentage BETWEEN 5 AND 15 THEN 'Medium Profit'
        WHEN MarginPercentage BETWEEN 0 AND 5 THEN 'Low Profit'
        ELSE 'Loss'
    END AS ProfitCategory,
    -- Recovery potential
    CASE 
        WHEN ProfitLossStatus = 'Loss' THEN ABS(Profit) * 0.7  -- 70% recoverable
        ELSE 0
    END AS RecoveryPotential
FROM Products;
GO

-- ============================================
-- 9. BUSINESS QUESTION: Price optimization opportunities
-- ============================================
WITH PriceAnalysis AS (
    SELECT 
        Category,
        AVG(Price) AS AvgPrice,
        AVG(MarginPercentage) AS AvgMargin,
        PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY Price) OVER (PARTITION BY Category) AS PriceQ1,
        PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY Price) OVER (PARTITION BY Category) AS PriceQ3
    FROM Products
    WHERE Profit > 0  -- Only look at profitable products for benchmark
    GROUP BY Category
)
SELECT DISTINCT
    p.Category,
    p.ProductName,
    p.Price,
    pa.AvgPrice AS CategoryAvgPrice,
    pa.PriceQ1 AS CategoryLowerRange,
    pa.PriceQ3 AS CategoryUpperRange,
    p.MarginPercentage,
    pa.AvgMargin AS CategoryAvgMargin,
    -- Pricing recommendation
    CASE 
        WHEN p.Price < pa.PriceQ1 THEN 'Price too low - Increase by 15%'
        WHEN p.Price > pa.PriceQ3 THEN 'Price too high - Check competition'
        WHEN p.MarginPercentage < pa.AvgMargin * 0.7 THEN 'Margin issue - Review costs'
        ELSE 'Price is optimal'
    END AS PricingRecommendation
FROM Products p
JOIN PriceAnalysis pa ON p.Category = pa.Category
WHERE p.ProfitLossStatus = 'Loss'  -- Focus on loss products
ORDER BY p.Category, p.Profit;

-- ============================================
-- 10. BUSINESS SUMMARY REPORT
-- ============================================
SELECT '📦 Total Products' AS Metric, FORMAT(COUNT(*), 'N0') AS Value FROM Products
UNION ALL
```
